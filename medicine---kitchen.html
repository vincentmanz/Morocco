library(dplyr)
library(hrbrthemes)
library(ggplot2)
library(viridis)

file <- read.csv("i-Marrakech_dataset.csv", sep=";")
data <- file %>% select(Sex,N_Pl_total, Freq_cooking, Freq_cooking_scale ) %>% 
  mutate(Sex, Sex = case_when(
    Sex == "F" ~ "Female",
    Sex == "M" ~ "Male"))

General overview

data %>%
  ggplot( aes(x=Sex, y=N_Pl_total, fill=Sex)) +
    geom_boxplot() +
    scale_fill_viridis(discrete = TRUE, alpha=0.6) +
    geom_jitter(color="black", size=0.9, alpha=0.9) +
    theme_ipsum() +
    theme(
      legend.position="none",
      plot.title = element_text(size=11)
    ) +
  ylab("Total Number of Plants") +
  xlab("sex") +
  ggtitle("Number of plants according to the gender - box plot")

Freq_cooking

data  %>% 
  ggplot(aes(x=Freq_cooking, y = N_Pl_total, color=Sex)) + 
  geom_point() + 
  xlab("Cooking frequency") + 
  ylab("Total Number of Plants") +
  theme_ipsum() +
  ggtitle("Number of plants according to the cooking frequency - scatter plot")
